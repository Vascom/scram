38 /// Convenience macro to throw SCRAM exceptions.
39 /// This is similar to BOOST_THROW_EXCEPTION;
40 /// however, it doesn't obfuscate
41 /// the resultant exception type to conform to boost::exception.
42 ///
43 /// @param[in] err The error type deriving from boost::exception.
44 #define SCRAM_THROW(err) \
45  throw err << ::boost::throw_function(BOOST_THROW_EXCEPTION_CURRENT_FUNCTION) \
46  << ::boost::throw_file(FILE_REL_PATH) \
47  << ::boost::throw_line(__LINE__)

55 /// The Error class is the base class
56 /// for all exceptions specific to the SCRAM code.
57 ///
58 /// @note The copy constructor is not noexcept as required by std::exception.
59 /// However, this class may only throw std::bad_alloc upon copy,
60 /// which may be produced anyway
61 /// even if the copy constructor were noexcept.
62 class Error : virtual public std::exception, virtual public boost::exception {
63  public:
64  /// Constructs a new error with a provided message.
65  ///
66  /// @param[in] msg The message to be passed with this error.
67  explicit Error(std::string msg) : msg_(std::move(msg)) {}
68 
69  /// @returns The formatted error message to be printed.
70  const char* what() const noexcept final { return msg_.c_str(); }
71 
72  private:
73  std::string msg_; ///< The error message.
74 };
